## VSEARCH-based pipeline for Illumina paired-end metabarcoding reads
## 11.2017 LH, VB.

##Get all fastq files into working directory and unzip them (my vsearch (v2.4.3_linux_x86_64, 15.6GB RAM, 8 cores)
##doesn't work with gzipped files, for some reason).

##Define Variables (Change to your primer sequences, these are MiniLGC):
PRIMER_F="TCATGGWACWGGWTGAACWGTWTAYCCYCC"
PRIMER_R_RC="TGRTTYTTTGGTCACCCTGAAGTTTACTCC"
PRIMER_R="GGAGTAAACTTCAGGGTGACCAAARAAYCA"
PRIMER_F_RC="GGRGGRTAWACWGTTCAWCCWGTWCCATGA"

##Merge paired-end reads:
for i in *_R1_*.fastq; do
	vsearch2 --fastq_mergepairs ${i} \
		--reverse ${i/_R1_/_R2_} \
		--fastqout ${i/R1_*/merged.fq} \
done

##Relabel the headers:
for i in $(ls *_merged.fq | cut -f 1-3 -d "_"); do sed -i "s/^@M/@${i}|/" ${i}_merged.fq; done

#Concatenate merged fastq's into one file:
cat *_merged.fq > allmerged.fq

#remove primers:
##forward:
cutadapt -g ^${PRIMER_F} -o trimmed1.fq allmerged.fq
##reverse:
cutadapt -a ${PRIMER_R_RC}$ -o trimmed2.fq trimmed1.fq
#rc:
cutadapt -g ^${PRIMER_R} -o trimmed3.fq trimmed2.fq
cutadapt -a ${PRIMER_F_RC}$ -o trimmed4.fq trimmed3.fq


##Quality filtering & sequence lengths:
vsearch2 --fastq_filter trimmed4.fq --fastq_minlen 300 --fastq_maxee 1 --fastaout reads.fa

##Dereplication #note: sizeout option important for subsequent de novo chimera removal step
vsearch2 --derep_fulllength reads.fa --sizeout --relabel Uniq --minuniquesize 2 --output uniques.fa

###Chimera removal:
vsearch2 --uchime_denovo uniques.fa --nonchimeras results.fa --abskew 1.5

########Cluster OTUs at 97%+ similarity.
vsearch2 -cluster_size results.fa --threads 8 --id 0.97 --iddef 0 --sizein --sizeout --relabel OTU --centroids otus.fa 

##Create OTU table: -the sample identifiers are apparently assumed to end at first non-alphanumeric or underscore char.
#Threshold of 97%?

#Convert allmerged fastq to fasta:
vsearch --fastq_filter allmerged.fq -fastaout allmerged.fa

#Edit it so that vsearch chops the headers only after the sample identifiers (will be the column names in table)
sed -i 's/|.*//' allmerged.fa
sed -i 's/_.*//' allmerged.fa
sed -i 's/-/_/g' allmerged.fa

##Finally, create table:
vsearch2 -usearch_global allmerged.fa -db otus.fa --id 0.97 --otutabout otu_table.txt
